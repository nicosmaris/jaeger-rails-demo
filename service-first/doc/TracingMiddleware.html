def call(env)
  result = nil
  OpenTracing.global_tracer.start_active_span(
    env["REQUEST_METHOD"] + ' ' + env["rack.url_scheme"] + '://' + env["HTTP_HOST"] + env["REQUEST_URI"],
    tags: {
      'component' => 'rails',
      'span.kind' => 'server',
      'http.method' => env["REQUEST_METHOD"],
      'http.original_url' => env["rack.url_scheme"] + '://' + env["HTTP_HOST"] + env["REQUEST_URI"],
      'request_id' => env["action_dispatch.request_id"]
    }
  ) do |scope|
    scope.span.set_tag('jaeger-debug-id', scope.span.context.trace_id)
    #OpenTracing.inject(span.context, OpenTracing::FORMAT_RACK, env)
    result = @app.call(env).tap do |status_code, _headers, _body|
    end
  end
  result
end
